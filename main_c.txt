#include "main.h"

#include "rpm.h"
#include <stdio.h>
#define BUTTON_GPIO_PORT   GPIOB
#define BUTTON_PIN         GPIO_PIN_2

ADC_HandleTypeDef hadc1;
I2C_HandleTypeDef hi2c2;
TIM_HandleTypeDef htim1;
TIM_HandleTypeDef htim2;

uint32_t adc_val = 0;
uint32_t duty    = 0;
uint32_t arr     = 0;

uint8_t dir_state = 0;
uint8_t  button_idle_level = 1;
uint8_t  button_pressed     = 0;
uint32_t last_button_time   = 0;
uint32_t last_lcd_update = 0;

void SystemClock_Config(void);
static void MX_GPIO_Init(void);
static void MX_ADC1_Init(void);
static void MX_TIM1_Init(void);
static void MX_TIM2_Init(void);
static void MX_I2C2_Init(void);

static void Button_UpdateState(void)
{
    uint32_t now   = HAL_GetTick();
    uint8_t  level = HAL_GPIO_ReadPin(BUTTON_GPIO_PORT, BUTTON_PIN);

    if ((level != button_idle_level) && (button_pressed == 0))
    {

        if ((now - last_button_time) > 50)
        {

            button_pressed   = 1;
            last_button_time = now;


            dir_state++;
            if (dir_state > 2)
                dir_state = 0;
        }
    }

    if (level == button_idle_level)
    {
        button_pressed = 0;
    }
}

int main(void)
{
  HAL_Init();
  SystemClock_Config();

  MX_GPIO_Init();
  MX_ADC1_Init();
  MX_TIM1_Init();
  MX_TIM2_Init();
  MX_I2C2_Init();

  Motor_Init();
  Motor_Standby(1);   // Active le TB6612FNG

  lcd_init();
  RPM_Init();

  HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1);

  arr = __HAL_TIM_GET_AUTORELOAD(&htim1);

  button_idle_level = HAL_GPIO_ReadPin(BUTTON_GPIO_PORT, BUTTON_PIN);
  last_button_time  = HAL_GetTick();
  last_lcd_update   = HAL_GetTick();

  while (1)
  {
    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, HAL_MAX_DELAY);
    adc_val = HAL_ADC_GetValue(&hadc1);
    HAL_ADC_Stop(&hadc1);
    duty = (adc_val * arr) / 4095u;

    if (duty < (arr / 40u))
        duty = 0;

    Button_UpdateState();


    switch (dir_state)
    {
        case 0: // STOP
            Motor_Standby(1);
            __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, 0);

            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_SET);    // rouge ON
            HAL_GPIO_WritePin(GPIOC, GPIO_PIN_7, GPIO_PIN_RESET);  // vert OFF
            HAL_GPIO_WritePin(GPIOB, GPIO_PIN_7, GPIO_PIN_RESET);  // bleu OFF
            break;

        case 1: // FWD
            Motor_Standby(1);
            Motor_SetDirection(1);
            __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, duty);

            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET);  // rouge OFF
            HAL_GPIO_WritePin(GPIOC, GPIO_PIN_7, GPIO_PIN_SET);    // vert ON
            HAL_GPIO_WritePin(GPIOB, GPIO_PIN_7, GPIO_PIN_RESET);  // bleu OFF
            break;

        case 2: // REV
            Motor_Standby(1);
            Motor_SetDirection(0);
            __HAL_TIM_SET_COMPARE(&htim1, TIM_CHANNEL_1, duty);

            HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET);  // rouge OFF
            HAL_GPIO_WritePin(GPIOC, GPIO_PIN_7, GPIO_PIN_RESET);  // vert OFF
            HAL_GPIO_WritePin(GPIOB, GPIO_PIN_7, GPIO_PIN_SET);    // bleu ON
            break;
    }

    uint32_t rpm = RPM_Get();

    uint32_t now = HAL_GetTick();
    if ((now - last_lcd_update) >= 200)
    {
        lcd_clear();

        lcd_set_cursor(0, 0);
        if (dir_state == 0)
            lcd_write_string("STATE: STOP");
        else if (dir_state == 1)
            lcd_write_string("STATE: FWD ");
        else
            lcd_write_string("STATE: REV ");

        char line[20];
        uint32_t pwm_percent = (arr == 0) ? 0 : (duty * 100u) / arr;
        sprintf(line, "PWM:%3lu%% RPM:%4lu", pwm_percent, rpm);

        lcd_set_cursor(0, 1);
        lcd_write_string(line);

        last_lcd_update = now;
    }

    HAL_Delay(5);

  }

}

static void MX_GPIO_Init(void)
{
  GPIO_InitTypeDef GPIO_InitStruct = {0};

  __HAL_RCC_GPIOF_CLK_ENABLE();
  __HAL_RCC_GPIOC_CLK_ENABLE();
  __HAL_RCC_GPIOA_CLK_ENABLE();
  __HAL_RCC_GPIOB_CLK_ENABLE();
  __HAL_RCC_GPIOE_CLK_ENABLE();

  /*Configure GPIO pin Output Level */
  HAL_GPIO_WritePin(GPIOE, GPIO_PIN_15, GPIO_PIN_RESET);

  /*Configure GPIO pin Output Level */
  HAL_GPIO_WritePin(GPIOB, GPIO_PIN_10|GPIO_PIN_11|GPIO_PIN_7, GPIO_PIN_RESET);

  /*Configure GPIO pin Output Level */
  HAL_GPIO_WritePin(GPIOC, GPIO_PIN_7, GPIO_PIN_RESET);

  /*Configure GPIO pin Output Level */
  HAL_GPIO_WritePin(GPIOA, GPIO_PIN_9, GPIO_PIN_RESET);

  /*Configure GPIO pins : PB0 PB1 (encodeur RPM) */
  GPIO_InitStruct.Pin = GPIO_PIN_0|GPIO_PIN_1;
  GPIO_InitStruct.Mode = GPIO_MODE_IT_RISING;
  GPIO_InitStruct.Pull = GPIO_PULLUP;
  HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);

  /*Configure GPIO pin : PE15 (STBY TB6612) */
  GPIO_InitStruct.Pin = GPIO_PIN_15;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
  HAL_GPIO_Init(GPIOE, &GPIO_InitStruct);

  /*Configure GPIO pins : PB10 PB11 PB7 (AIN1, AIN2, LED bleue) */
  GPIO_InitStruct.Pin = GPIO_PIN_10|GPIO_PIN_11|GPIO_PIN_7;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
  HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);

  /*Configure GPIO pin : PC7 (LED verte) */
  GPIO_InitStruct.Pin = GPIO_PIN_7;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
  HAL_GPIO_Init(GPIOC, &GPIO_InitStruct);

  /*Configure GPIO pin : PA9 (LED rouge) */
  GPIO_InitStruct.Pin = GPIO_PIN_9;
  GPIO_InitStruct.Mode = GPIO_MODE_OUTPUT_PP;
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
  HAL_GPIO_Init(GPIOA, &GPIO_InitStruct);

  /* PB2: bouton direction/STOP (entr√©e) */
  GPIO_InitStruct.Pin = GPIO_PIN_2;
  GPIO_InitStruct.Mode = GPIO_MODE_INPUT;
  GPIO_InitStruct.Pull = GPIO_PULLUP;
  HAL_GPIO_Init(GPIOB, &GPIO_InitStruct);

  /* PE9: TIM1_CH1 (PWM moteur) */
  GPIO_InitStruct.Pin = GPIO_PIN_9;
  GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
  GPIO_InitStruct.Pull = GPIO_NOPULL;
  GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_LOW;
  GPIO_InitStruct.Alternate = GPIO_AF1_TIM1;
  HAL_GPIO_Init(GPIOE, &GPIO_InitStruct);

  /* EXTI interrupt init*/
  HAL_NVIC_SetPriority(EXTI0_IRQn, 0, 0);
  HAL_NVIC_EnableIRQ(EXTI0_IRQn);

  HAL_NVIC_SetPriority(EXTI1_IRQn, 0, 0);
  HAL_NVIC_EnableIRQ(EXTI1_IRQn);

}

{
    if (GPIO_Pin == GPIO_PIN_0 || GPIO_Pin == GPIO_PIN_1)
    {
        RPM_EncoderPulse();
    }
}
